## MVP-FIRST APPROACH — BUILD FRONTEND WITHOUT BACKEND

**All phases must be built frontend-first using mock data. Do not wait for a backend.**

### Core Rule
Every service method returns mock data via `of(mockValue)` from RxJS. The component code is identical whether talking to a real API or mock — only the service internals change. When the backend is ready, swap `of(mockData)` for `this.api.get(...)` in the service. Nothing else changes.

### Mock Service Pattern

```typescript
// features/booking/services/booking.service.ts
import { Injectable, inject } from '@angular/core';
import { Observable, of } from 'rxjs';
import { delay } from 'rxjs/operators';
import { ApiService } from '@core/services/api.service';
import { Booking, CreateBookingRequest } from '../models/booking.model';
import { MOCK_BOOKINGS } from '../mocks/booking.mocks';
import { environment } from '@environments/environment';

@Injectable({ providedIn: 'root' })
export class BookingService {
  private api = inject(ApiService);

  getBookings(): Observable<Booking[]> {
    if (environment.useMocks) {
      return of(MOCK_BOOKINGS).pipe(delay(400)); // simulate network delay
    }
    return this.api.get<Booking[]>('/api/v1/bookings');
  }

  createBooking(payload: CreateBookingRequest): Observable<Booking> {
    if (environment.useMocks) {
      const newBooking: Booking = { ...MOCK_BOOKINGS[0], id: crypto.randomUUID(), ...payload };
      return of(newBooking).pipe(delay(600));
    }
    return this.api.post<Booking>('/api/v1/bookings', payload);
  }
}
```

## DESIGN SYSTEM

### Design Direction

The clinic system uses a **clean, professional medical aesthetic**: trustworthy, calm, and precise. Think modern healthcare SaaS — not sterile hospital forms, but a polished web app a doctor would be proud to use. 

Key qualities: structured whitespace, clear visual hierarchy, confident typography, and purposeful color use. Nothing decorative that does not serve clarity.

## API SERVICE WRAPPER

```typescript
// core/services/api.service.ts
@Injectable({ providedIn: 'root' })
export class ApiService {
  private readonly baseUrl = environment.apiUrl;

  constructor(private http: HttpClient) {}

  get<T>(endpoint: string, params?: HttpParams): Observable<T> {
    return this.http.get<T>(`${this.baseUrl}${endpoint}`, { params });
  }

  post<T>(endpoint: string, body: unknown): Observable<T> {
    return this.http.post<T>(`${this.baseUrl}${endpoint}`, body);
  }

  put<T>(endpoint: string, body: unknown): Observable<T> {
    return this.http.put<T>(`${this.baseUrl}${endpoint}`, body);
  }

  delete<T>(endpoint: string): Observable<T> {
    return this.http.delete<T>(`${this.baseUrl}${endpoint}`);
  }
}
```

Feature services inject `ApiService`, never `HttpClient` directly. In MVP mode, services return `of(mockData)` — same return type, same interface.

## HTTP INTERCEPTORS

### auth.interceptor.ts
- Attaches `Authorization: Bearer {accessToken}` to every outgoing request
- Skips public endpoints (login, register, public doctors, public services, announcements)
- In MVP mode: still runs, token is a fake string from AuthStore — harmless

```typescript
export const authInterceptor: HttpInterceptorFn = (req, next) => {
  const authService = inject(AuthService);
  const token = authService.getAccessToken();

  if (token) {
    req = req.clone({
      setHeaders: { Authorization: `Bearer ${token}` }
    });
  }

  return next(req);
};
```

### refresh.interceptor.ts
- Intercepts `401 Unauthorized` responses
- Calls `POST /api/v1/auth/refresh` with the stored refresh token
- If refresh succeeds: retries original request with new access token
- If refresh fails: logs user out, redirects to login
- In MVP mode: 401 never fires (mocks don't make real HTTP calls)

### NgRx SignalStore — global shared state

```typescript
// core/stores/auth.store.ts
export const AuthStore = signalStore(
  { providedIn: 'root' },
  withState({
    user: null as AuthUser | null,
    accessToken: null as string | null,
    isAuthenticated: false,
  }),
  withMethods((store, authService = inject(AuthService)) => ({
    login: rxMethod<LoginRequest>(
      pipe(
        switchMap(credentials => authService.login(credentials)),
        tapResponse({
          next: (response) => patchState(store, {
            user: response.user,
            accessToken: response.accessToken,
            isAuthenticated: true
          }),
          error: console.error
        })
      )
    ),
    logout() {
      patchState(store, { user: null, accessToken: null, isAuthenticated: false });
    }
  }))
);
```

Global stores:
- `AuthStore` — current user, tokens, role
- `NotificationStore` — unread count, notification list
- `ClinicSettingsStore` — clinic name, logo, colors (loaded once on app start)

## AUTH GUARDS

```typescript
// core/guards/auth.guard.ts
export const AuthGuard: CanActivateFn = () => {
  const authStore = inject(AuthStore);
  const router = inject(Router);

  if (!authStore.isAuthenticated()) {
    router.navigate(['/auth/login']);
    return false;
  }
  return true;
};

// core/guards/role.guard.ts
export const RoleGuard: CanActivateFn = (route) => {
  const authStore = inject(AuthStore);
  const router = inject(Router);
  const requiredRoles = route.data['roles'] as string[];

  if (!requiredRoles.includes(authStore.user()?.role ?? '')) {
    router.navigate(['/unauthorized']);
    return false;
  }
  return true;
};

// core/guards/first-login.guard.ts
export const FirstLoginGuard: CanActivateFn = () => {
  const authStore = inject(AuthStore);
  const router = inject(Router);

  if (authStore.user()?.isFirstLogin) {
    router.navigate(['/auth/set-password']);
    return false;
  }
  return true;
};
```

```typescript
createBooking(payload: CreateBookingRequest): Observable<Booking> {
```
